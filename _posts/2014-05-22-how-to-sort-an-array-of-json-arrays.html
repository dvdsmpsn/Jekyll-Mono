Here's a simple unsorted array of JSON objects:

[code]
var someArray = [
{
"id": 2,
"name": "gamma"
},
{
"id": 3,
"name": "alpha"
},
{
"id": 1,
"name": "beta"
}
];
[/code]

And a generic sorting function:

[code]
/**
* Generic array sorting
*
* @param property
* @returns {Function}
*/
var sortByProperty = function (property) {
return function (x, y) {
return ((x[property] === y[property]) ? 0 : ((x[property] > y[property]) ? 1 : -1));
};
};
[/code]

Calling this...
[code]
someArray.sort(sortByProperty('id'));
[/code]

...produces an array sorted by the id property:
[code]

[
{
"id": 1,
"name": "beta"
},
{
"id": 2,
"name": "gamma"
},
{
"id": 3,
"name": "alpha"
}
]
[/code]

Calling this...
[code]
someArray.sort(sortByProperty('name'));
[/code]

...produces an array sorted by the name property:
[code]
[
{
"id": 3,
"name": "alpha"
},
{
"id": 1,
"name": "beta"
},
{
"id": 2,
"name": "gamma"
}
]
[/code]

Simple.
