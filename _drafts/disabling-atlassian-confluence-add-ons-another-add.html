A simple code snippet for when you need to disable a pile of Confluence add-ons from within your own custom plugin.


String[] ADDONS_TO_DISABLE = {

	"com.atlassian.confluence.plugins.confluence-mentions-plugin"

	// Add more plugins keys here as required

};


for (String pluginKey : ADDONS_TO_DISABLE)

{

    if (pluginAccessor.isPluginEnabled(pluginKey))

    {

        pluginController.disablePlugin(pluginKey);

        log.info(" ++ Disabling plugin: " + pluginKey);

    }

    else

    {

        log.info(" ++ Plugin already disabled: " + pluginKey);

    }

}
